from sklearn.ensemble     import RandomForestRegressor
from sklearn.linear_model import LinearRegression
from sklearn.metrics      import r2_score

#for Reg, Reg_name in zip([LinearRegression, RandomForestRegressor], ["Lineare Regression", "Random Forest"]):

r2_train  = dict()
r2_val    = dict()
std_train = dict()
std_val   = dict()

for trainsize in range(10,433, 10):
    X_train = df.iloc[:trainsize,:-1]
    y_train = df.iloc[:trainsize,-1]

    X_val = df.iloc[trainsize:,:-1]
    y_val = df.iloc[trainsize:,-1]

    model = RandomForestRegressor( max_features=5, n_estimators=100, max_depth=5 )
    model.fit ( X_train, y_train )
    
    r2_train[trainsize] = r2_score ( y_train, model.predict ( X_train ) )
    r2_val  [trainsize] = r2_score ( y_val,   model.predict ( X_val   ) )
    
    std_train[trainsize] = pd.DataFrame ( [m.predict(X_train.values) for m in model.estimators_] ).std(axis=0).mean()
    std_val  [trainsize] = pd.DataFrame ( [m.predict(X_val.values  ) for m in model.estimators_] ).std(axis=0).mean()

Interpretation: Fr sehr wenig Trainingsdaten (<100) ist das Modell instabil (d.h. das Modell ist zu sehr von Zufallseinflssen geprgt). Fr sehr wenig Validierungsdaten (Trainingsdaten > 350) wird die Evaluierung instabil (d.h. nicht vertrauenswrdig). Dazwischen verhlt es sich berraschend stabil.

Mit zunehmender Trainingsdatengre nimmt die Tendenz zum Overfitten ab.

Im rechten Plot sieht man, dass die Standardabweichung in den Vorhersagen (auf Trainings und Validierungsdaten) gleichermaen geringer wird, wenn mehr Trainingsdaten verwendet wurden. D.h. das Modell wird sich sicherer (aber nicht notwendigerweise besser, wie auf dem Plateau zwischen 100 und 300 im linken Plot in der Validierungskurve zu sehen ist) in der Vorhersage.
