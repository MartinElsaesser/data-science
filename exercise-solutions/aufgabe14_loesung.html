R = np.random.default_rng ( 42 )
x = R.normal ( loc=0.0, scale=1.0, size=1000 )
y = R.normal ( loc=0.0, scale=1.0, size=1000 )

g = sns.jointplot ( x=x, y=y,
                    marginal_kws={"bins":50,              # erzwinge ein feineres Binning 
                                  "stat":"density"} )     # normiere Histogramm-Flche auf 100%,
                                                          # um sie vergleichbar zu machen mit der
                                                          # Normalverteilungsdichte spter
g.ax_joint.set_xlabel('x')
g.ax_joint.set_ylabel('y')

xrange = np.linspace(-3,3,100)
g.ax_marg_x.plot ( xrange, np.exp(-xrange**2/2)/np.sqrt(2*np.pi), color="red" )
g.ax_marg_y.plot ( np.exp(-xrange**2/2)/np.sqrt(2*np.pi), xrange, color="red" )

cors_pearson  = dict()
cors_spearman = dict()

for n in range(0,1001,1):
    df_temp.iloc[0,:] = n
    
    cors_pearson[n]  = df_temp.corr(method="pearson" ).loc["x","y"]
    cors_spearman[n] = df_temp.corr(method="spearman").loc["x","y"]
