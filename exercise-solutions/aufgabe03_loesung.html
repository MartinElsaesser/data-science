  1. Schreiben Sie eine Funktion, die den Heart-Datensatz in drei gleich groe Teile teilt. Die Zuordnung der Zeilen zu den jeweiligen Teilen soll zufllig geschehen. Recherchieren Sie hierzu z.B. numpy.random.permutation(). Die drei Teilmengen werden im folgenden Trainings-, Validierungs- und Testset genannt.

def split_into_thirds ( df, random_seed = 42 ):
    # use a random seed to make the shuffle reproducible
    R = np.random.default_rng(42)
    df_shuffled = df.iloc [ R.permutation ( df.index ), : ]
    
    # split into thirds
    thirdlength = len(df)//3
    df_train = df_shuffled.iloc[:thirdlength, :]
    df_val   = df_shuffled.iloc[thirdlength:2*thirdlength, :]
    df_test  = df_shuffled.iloc[2*thirdlength:, :]
    
    return df_train, df_val, df_test

def calculate_metrics ( y, pred, silent=False, headline="" ):
    TP = (y==1) & (pred == 1)
    FP = (y==0) & (pred == 1)
    TN = (y==0) & (pred == 0)
    FN = (y==1) & (pred == 0)

    TP = TP.sum()
    FP = FP.sum()
    TN = TN.sum()
    FN = FN.sum()
    acc = (TP+TN)/(TP+TN+FP+FN)
    
    if not silent:
        print("----------------")
        if not headline=="": print(headline)
        print(f"TP: {TP}, FP: {FP}, TN: {TN}, FN: {FN}")
        print(f"Accuracy: {acc*100:.02f}%")
        print("----------------")
    
    return {"TP":TP,"FP":FP,"TN":TN,"FN":FN,"acc":acc}
    # Especially if there are many return values:
    # It is often convenient not to return an (unnamed) list;
    # instead return a dictionary to store the information
    # which return value has which meaning

Es gengt nicht, die Performance auf den Validierungsdaten zu kennen, da man whrend des Hyperparameter-Tunings diesen sehr oft evaluiert hat. Es besteht die Gefahr, dass aus Zufallsgrnden ein Satz an Hyperparametern gut aussieht ("multiples testen"). Die Generalisierungsfhigkeit wird durch die einmalige Evaluation auf einem Testdatensatz bewertet.
