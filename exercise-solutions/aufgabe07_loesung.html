def split_into_thirds ( df, random_seed = 42, stratify_column = None ):
    # use a random seed to make the shuffle reproducible
    R = np.random.default_rng(42)
    
    # shuffle
    df_shuffled = df.loc [ R.permutation ( df.index ), : ]
    
    if not stratify_column is None:
        df0 = df_shuffled[ df_shuffled[stratify_column]==0 ]
        df1 = df_shuffled[ df_shuffled[stratify_column]==1 ]
        to_split = [df0, df1]
    else:
        to_split = [df_shuffled]
    
    df_train = []
    df_val   = []
    df_test  = []
    
    for dfs in to_split:
        # split into thirds
        thirdlength = len(dfs)//3
        df_train.append( dfs.iloc[:thirdlength, :] )
        df_val.append( dfs.iloc[thirdlength:2*thirdlength, :] )
        df_test.append( dfs.iloc[2*thirdlength:, :] )
    
    df_train = pd.concat ( df_train, axis=0 )
    df_val = pd.concat ( df_val, axis=0 )
    df_test = pd.concat ( df_test, axis=0 )
    
    return df_train, df_val, df_test

from sklearn.preprocessing import OneHotEncoder
enc = OneHotEncoder( handle_unknown="ignore", sparse=False, drop="if_binary" )

# handle_unknown="ignore" => any unknown category will be mapped to "all zeros"
# drop="if_binary"        => Sex is a binary variable (it contains only "M" and "F"), so it is sufficient
#                            to store if Sex_M is 0 or 1. More generally, if there are n categories,
#                            it is sufficient to create n-1 columns.
#                            Note however that this collides with handle_unknown="ignore".
#                            Namely, if a third value is encountered during testing (e.g. "unknown" for Sex),
#                            it will be mapped to 0 (because of handle_unknown="ignore")
#                            which will be interpreted as "Sex=F" (because of drop="if_binary").
#                            The drop-parameter is used here for demonstration only.

X_train_cat = pd.DataFrame ( enc.fit_transform ( X_train.loc[:,catcols] ),
                             columns=enc.get_feature_names_out(),

import numpy as np
xrange = np.linspace(0.01,0.99,99)

import matplotlib.pyplot as plt
f = lambda x: 2*x*(1-x)
g = lambda x: -x*np.log(x)-(1-x)*np.log(1-x)

plt.plot ( xrange, f(xrange), label="Gini" )
plt.plot ( xrange, f(0.5)/g(0.5)*g(xrange), label="Entropie (auf gleiche Hhe normiert)" )
plt.legend();
